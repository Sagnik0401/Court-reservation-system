-- Tennis Court Reservation System - Database Setup
-- Run this in your Supabase SQL Editor

-- 1. CREATE COURTS TABLE
CREATE TABLE courts (
  id SERIAL PRIMARY KEY,
  name VARCHAR(50) NOT NULL,
  surface_type VARCHAR(20) DEFAULT 'Clay',
  is_active BOOLEAN DEFAULT true,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW()
);

-- 2. CREATE BOOKINGS TABLE
CREATE TABLE bookings (
  id SERIAL PRIMARY KEY,
  court_id INTEGER REFERENCES courts(id) ON DELETE RESTRICT,
  member_id INTEGER REFERENCES members(id) ON DELETE CASCADE,
  booking_date DATE NOT NULL,
  start_time TIME NOT NULL,
  duration_minutes INTEGER NOT NULL CHECK (duration_minutes IN (30, 60, 90, 120)),
  status VARCHAR(20) DEFAULT 'confirmed' CHECK (status IN ('pending', 'confirmed', 'cancelled', 'completed')),
  notes TEXT,
  created_at TIMESTAMPTZ DEFAULT NOW(),
  updated_at TIMESTAMPTZ DEFAULT NOW(),
  
  -- Ensure no double bookings for same court, date, and time
  CONSTRAINT unique_court_datetime UNIQUE (court_id, booking_date, start_time)
);

-- 3. CREATE INDEXES FOR PERFORMANCE
CREATE INDEX idx_bookings_court_date ON bookings(court_id, booking_date);
CREATE INDEX idx_bookings_member ON bookings(member_id);
CREATE INDEX idx_bookings_status ON bookings(status);
CREATE INDEX idx_bookings_datetime ON bookings(booking_date, start_time);

-- 9. CREATE A FUNCTION TO CHECK BOOKING CONFLICTS
CREATE OR REPLACE FUNCTION check_booking_conflict(
  p_court_id INTEGER,
  p_booking_date DATE,
  p_start_time TIME,
  p_duration_minutes INTEGER,
  p_exclude_booking_id INTEGER DEFAULT NULL
) RETURNS BOOLEAN AS $$
DECLARE
  p_end_time TIME;
  conflict_count INTEGER;
BEGIN
  -- Calculate end time
  p_end_time := p_start_time + (p_duration_minutes || ' minutes')::INTERVAL;
  
  -- Check for conflicts
  SELECT COUNT(*) INTO conflict_count
  FROM bookings
  WHERE court_id = p_court_id
    AND booking_date = p_booking_date
    AND status IN ('confirmed', 'pending')
    AND (id != p_exclude_booking_id OR p_exclude_booking_id IS NULL)
    AND (
      -- New booking starts during existing booking
      (p_start_time >= start_time AND p_start_time < start_time + (duration_minutes || ' minutes')::INTERVAL)
      OR
      -- New booking ends during existing booking  
      (p_end_time > start_time AND p_end_time <= start_time + (duration_minutes || ' minutes')::INTERVAL)
      OR
      -- New booking completely encompasses existing booking
      (p_start_time <= start_time AND p_end_time >= start_time + (duration_minutes || ' minutes')::INTERVAL)
    );
    
  RETURN conflict_count > 0;
END;
$$ LANGUAGE plpgsql;

-- 10. CREATE A FUNCTION TO GET AVAILABLE SLOTS
CREATE OR REPLACE FUNCTION get_available_slots(
  p_court_id INTEGER,
  p_booking_date DATE
) RETURNS TABLE(
  slot_time TIME,
  is_available BOOLEAN,
  booking_id INTEGER
) AS $$
DECLARE
  slot_start TIME;
  morning_slots TIME[] := ARRAY['06:00', '06:30', '07:00', '07:30', '08:00', '08:30', '09:00', '09:30', '10:00', '10:30', '11:00', '11:30'];
  evening_slots TIME[] := ARRAY['15:00', '15:30', '16:00', '16:30', '17:00', '17:30', '18:00', '18:30', '19:00', '19:30', '20:00', '20:30', '21:00', '21:30', '22:00'];
  all_slots TIME[];
  slot_record RECORD;
BEGIN
  -- Combine morning and evening slots
  all_slots := morning_slots || evening_slots;
  
  -- Check each slot
  FOREACH slot_start IN ARRAY all_slots LOOP
    SELECT b.id INTO slot_record
    FROM bookings b
    WHERE b.court_id = p_court_id
      AND b.booking_date = p_booking_date
      AND b.status IN ('confirmed', 'pending')
      AND slot_start >= b.start_time
      AND slot_start < b.start_time + (b.duration_minutes || ' minutes')::INTERVAL
    LIMIT 1;
    
    slot_time := slot_start;
    is_available := slot_record.id IS NULL;
    booking_id := slot_record.id;
    
    RETURN NEXT;
  END LOOP;
END;
$$ LANGUAGE plpgsql;
